749 void draw_vertex(index_t v) {
750 if(do_animation_) {
751 if(mesh_->vertices.single_precision()) {
752 const GLUPfloat* p =
753 mesh_->vertices.single_precision_point_ptr(v);
754 float t = float(time_);
755 float s = 1.0f - float(time_);
756 glupPrivateVertex3f(
757 s*p[0] + t*p[3],
758 s*p[1] + t*p[4],
759 s*p[2] + t*p[5]
760 );
761 } else {
762 const GLUPdouble* p = mesh_->vertices.point_ptr(v);
763 double s = 1.0 - time_;
764 glupPrivateVertex3d(
765 s*p[0] + time_*p[3],
766 s*p[1] + time_*p[4],
767 s*p[2] + time_*p[5]
768 );
769 }
770 } else {
771 if(mesh_->vertices.single_precision()) {
772 if(mesh_->vertices.dimension() < 3) {
773 glupPrivateVertex2fv(
774 mesh_->vertices.single_precision_point_ptr(v)
775 );
776 } else {
777 glupPrivateVertex3fv(
778 mesh_->vertices.single_precision_point_ptr(v)
779 );
780 }
781 } else {
782 if(mesh_->vertices.dimension() < 3) {
783 glupPrivateVertex2dv(
784 mesh_->vertices.point_ptr(v)
785 );
786 } else {
787 glupPrivateVertex3dv(
788 mesh_->vertices.point_ptr(v)
789 );
790 }
791 }
792 }

1028 void draw_sequences(
1029 const MeshSubElementsStore& elements,
1030 std::function<void(index_t, index_t)> draw
1031 ) {
1032 // GLUP hardware primitive filtering.
1033 // Not supported in all profiles.
1034 // If supported, primitives are filtered by the GPU,
1035 // else, they are filtered here explicitly.
1036 const bool hw_filtering = hw_filtering_supported();
1037
1038 Filter* filter = nullptr;
1039 if(&elements == &mesh_->vertices) {
1040 filter = &vertices_filter_;
1041 } else if(&elements == &mesh_->facets) {
1042 filter = &facets_filter_;
1043 } else if(&elements == &mesh_->cells) {
1044 filter = &cells_filter_;
1045 }
1046 geo_assert(filter != nullptr) ;
1047
1048 filter->begin(elements.attributes(), hw_filtering);
1049 index_t e = 0;
1050 while(e < elements.nb()) {
1051 while(e<elements.nb() && !(hw_filtering||filter->test(e))) {
1052 ++e;
1053 }
1054 if(e<elements.nb()) {
1055 index_t begin = e;
1056 while(e<elements.nb() && (hw_filtering||filter->test(e))) {
1057 ++e;
1058 }
1059 index_t end = e;
1060 glupBasePickingId(GLUPuint64(begin));
1061 draw(begin,end);
1062 }
1063 }
1064 filter->end();
1065 glupBasePickingId(0);
1066 }

1082 void draw_sequences_if(
1083 const MeshSubElementsStore& elements,
1084 std::function<bool(index_t)> predicate,
1085 std::function<void(index_t, index_t)> draw
1086 ) {
1087 // GLUP hardware primitive filtering.
1088 // Not supported in all profiles.
1089 // If supported, primitives are filtered by the GPU,
1090 // else, they are filtered here explicitly.
1091 const bool hw_filtering = hw_filtering_supported();
1092
1093 Filter* filter = nullptr;
1094 if(&elements == &mesh_->vertices) {
1095 filter = &vertices_filter_;
1096 } else if(&elements == &mesh_->facets) {
1097 filter = &facets_filter_;
1098 } else if(&elements == &mesh_->cells) {
1099 filter = &cells_filter_;
1100 }
1101 geo_assert(filter != nullptr) ;
1102 filter->begin(elements.attributes(), hw_filtering);
1103
1104 index_t e = 0;
1105 while(e < elements.nb()) {
1106 while(
1107 e<elements.nb() && (
1108 !(hw_filtering || filter->test(e)) || !predicate(e))
1109 ) {
1110 ++e;
1111 }
1112 if(e<elements.nb()) {
1113 index_t begin = e;
1114 while(
1115 e<elements.nb() &&
1116 (hw_filtering || filter->test(e)) &&
1117 predicate(e)
1118 ) {
1119 ++e;
1120 }
1121 index_t end = e;
1122 glupBasePickingId(GLUPuint64(begin));
1123 draw(begin,end);
1124 }
1125 }
1126 filter->end();
1127 glupBasePickingId(0);
