287 bool edge_is_intersected(index_t f, index_t lv, index_t config) {
288 index_t v1 = origin_vertex(f, lv);
289 index_t v2 = destination_vertex(f, lv);
290 bool v1_in = ((config & 1u<<v1) != 0);
291 bool v2_in = ((config & 1u<<v2) != 0);
