981 bool radial_sort(index_t bndl, RadialSort& RS) {
982 geo_debug_assert(bndl < nb());

984 return true;

987 auto e = H_.begin() + std::ptrdiff_t(bndl_start_[bndl+1]);
988 RS.init(*b);
989 std::sort(b, e, RS);
990 bool OK = !RS.degenerate();
991 bndl_is_sorted_[bndl] = OK;
992 return OK;
